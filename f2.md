# Git Branching CLI Commands

# Create a branch without switching
1. git branch feature1

# Create and switch to created branch
1. git checkout -b feature2
2. git switch -c feature2

# Delete a branch 
1. git branch -d <branch name>
2. git branch -D <branch name>
# Merge a branch
1. git merge <branch name>
# Rename a Branch
1. git branch -m <new branch>
2. git branch -M <new branch>
